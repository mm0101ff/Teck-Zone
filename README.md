<script>
  const btnAbout = document.getElementById('btn-about');
  const btnSkills = document.getElementById('btn-skills');
  const btnContact = document.getElementById('btn-contact');

  const aboutSection = document.getElementById('about');
  const skillsSection = document.getElementById('skills');
  const contactSection = document.getElementById('contact');

  const buttons = [btnAbout, btnSkills, btnContact];
  const sections = [aboutSection, skillsSection, contactSection];

  function showSection(index) {
    sections.forEach((sec, i) => {
      sec.style.display = i === index ? 'block' : 'none';
    });
    buttons.forEach((btn, i) => {
      btn.classList.toggle('active', i === index);
    });
  }

  btnAbout.addEventListener('click', () => showSection(0));
  btnSkills.addEventListener('click', () => showSection(1));
  btnContact.addEventListener('click', () => showSection(2));
</script>
